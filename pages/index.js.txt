// pages/index.js

import { useEffect } from "react";

export default function Home() {
  // Using vanilla JavaScript for interactivity
  useEffect(() => {
    const signInButton = document.getElementById("sign-in-button");
    const signOutButton = document.getElementById("sign-out-button");
    const userName = document.getElementById("user-name");
    const userEmail = document.getElementById("user-email");

    signInButton.addEventListener("click", () => {
      // Simulate sign-in with vanilla JS (no React needed)
      const user = {
        displayName: "User Name",
        email: "user@example.com",
        photoURL: "https://randomuser.me/api/portraits/men/10.jpg",
      };

      // Update the UI with the user's info
      userName.textContent = user.displayName;
      userEmail.textContent = user.email;
      document.getElementById("user-info").classList.remove("hidden");
      document.getElementById("sign-in-button").classList.add("hidden");
    });

    signOutButton.addEventListener("click", () => {
      // Simulate sign-out
      userName.textContent = "";
      userEmail.textContent = "";
      document.getElementById("user-info").classList.add("hidden");
      document.getElementById("sign-in-button").classList.remove("hidden");
    });
  }, []);

  return (
    <div className="flex flex-col items-center justify-center min-h-screen bg-gray-100">
      <div id="user-info" className="hidden bg-white p-6 rounded-lg shadow-md text-center">
        <img src="" alt="User" id="user-photo" className="rounded-full w-16 h-16 mx-auto mb-4" />
        <h2 id="user-name" className="text-lg font-semibold"></h2>
        <p id="user-email" className="text-gray-600"></p>
        <button
          id="sign-out-button"
          className="mt-4 px-4 py-2 bg-red-500 text-white rounded-lg hover:bg-red-600"
        >
          Sign Out
        </button>
      </div>

      <button
        id="sign-in-button"
        className="px-6 py-3 bg-blue-500 text-white rounded-lg text-lg font-semibold hover:bg-blue-600"
      >
        Sign in with Google
      </button>
    </div>
  );
}
